\chapter*{Resumo}
\label{cap:resumo}%%
\thispagestyle{empty}

\PARstartLetter{S}{Sistemas}\altc{simples de gerenciamento de cadastros,
vendas, pagamentos e entrega de produtos, podem atender pequenas empresas no ramo de comércio de produtos alimentícios de forma rápida, moderna e baixo custo.Sistemas assim possuem uma arquitetura bem distribuída que podem atender um amplo público em diversas localidades e pode ser executado de vários dispositivos modernos como em smartphones por exemplo, não perdendo seu desempenho com o aumento do número de usuários.Dessa forma possibilitando um baixo custo ao fazer a compra do sistema}

\altc{Este trabalho tem como objetivo mostrar o desenvolvimento de um sistema baseado na Arquitetura de software orientado a aplicações em Font-End, com o foco em um gerenciador de vendas, pagamentos e entregas de produtos de uma Doceria.Apresento também as tecnologias, padrões e modelos utilizados durante o desenvolvimento.}
